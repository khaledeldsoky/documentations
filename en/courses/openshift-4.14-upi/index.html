vSphere Environment

Download RHCOS OVA
WSL
export RHCOS_VERSION=4.14.0
cd $OCP4_DIR/rhcos
wget https://mirror.openshift.com/pub/openshift-v4/dependencies/rhcos/4.14/${RHCOS_VERSION}/rhcos-${RHCOS_VERSION}-x86_64-vmware.x86_64.ova
sha256sum rhcos-${RHCOS_VERSION}-x86_64-vmware.x86_64.ova
Import RHCOS OVA as Template
WSL
govc import.ova \
  -name rhcos-4.14-template \
  -ds hdd \
  -pool /Datacenter/host/Openshift/Resources \
  -folder /Datacenter/vm/ \
  $OCP4_DIR/rhcos/rhcos-4.14.0-x86_64-vmware.x86_64.ova

govc vm.markastemplate rhcos-4.14-template
Verify

govc vm.info rhcos-4.14-template shows the VM as a template .

HAProxy + NFS VM

Create VM from ISO
WSL
govc vm.create \
  -name nfs-haproxy \
  -m 8192 -c 4 \
  -disk "40G" \
  -disk.controller pvscsi \
  -net "VM Network" \
  -pool /Datacenter/host/Openshift/Resources \
  -folder /Datacenter/vm/ \
  -on false

govc vm.dvd.insert \
  -vm nfs-haproxy \
  -ds hdd \
  /hdd/isos/CentOS-Stream-9-latest-x86_64-dvd1.iso

govc vm.disk.create \
  -vm nfs-haproxy \
  -name nfs-haproxy/nfs-data \
  -size "100G" -ds hdd -thin

govc vm.power -on nfs-haproxy
govc vm.console nfs-haproxy
⚠
Open the console URL in a browser to install CentOS 9 manually . Set hostname nfs-haproxy.ocp4.example.com and user khaled .
Configure Static IP

Run on the nfs-haproxy VM via SSH .

nfs-haproxy
ssh -i ~/.ssh/openshift khaled@172.20.0.210

sudo nmcli con mod ens192 \
  ipv4.addresses 172.20.0.210/16 \
  ipv4.gateway 172.20.0.254 \
  ipv4.dns 172.16.6.70 \
  ipv4.dns-search ocp4.example.com \
  ipv4.method manual

sudo nmcli con down ens192 && sudo nmcli con up ens192
ip addr show ens192
Install Packages
nfs-haproxy
sudo dnf install -y haproxy nfs-utils nginx firewalld policycoreutils-python-utils
sudo systemctl enable --now haproxy nfs-server nginx firewalld
Configure HAProxy

Replace /etc/haproxy/haproxy.cfg on nfs-haproxy VM .

nfs-haproxy — /etc/haproxy/haproxy.cfg
global
    log /dev/log local0
    chroot /var/lib/haproxy
    stats socket /run/haproxy/admin.sock mode 660
    user haproxy group haproxy
    daemon

defaults
    log global
    mode tcp
    option tcplog
    timeout connect 10s
    timeout client 30s
    timeout server 30s

frontend openshift-api
    bind *:6443
    default_backend openshift-api

backend openshift-api
    balance source
    server bootstrap 172.20.0.230:6443 check
    server master-0 172.20.0.231:6443 check
    server master-1 172.20.0.232:6443 check
    server master-2 172.20.0.233:6443 check

frontend mcs
    bind *:22623
    default_backend mcs

backend mcs
    balance source
    server bootstrap 172.20.0.230:22623 check
    server master-0 172.20.0.231:22623 check
    server master-1 172.20.0.232:22623 check
    server master-2 172.20.0.233:22623 check

frontend openshift-ingress
    bind *:443
    bind *:80
    default_backend openshift-ingress

backend openshift-ingress
    balance source
    server worker-0 172.20.0.234:443 check
    server worker-1 172.20.0.235:443 check
nfs-haproxy — Apply config
sudo cp /etc/haproxy/haproxy.cfg "$(date +%Y%m%d-%H%M)".bak
# Paste config content above, then:
sudo haproxy -c -f /etc/haproxy/haproxy.cfg
sudo systemctl restart haproxy
sudo ss -tlnp | grep -E ':(6443|22623|443|80)'
Verify

sudo haproxy -c -f /etc/haproxy/haproxy.cfg — Configuration file is valid

sudo ss -tlnp | grep haproxy shows 4 listening ports .

Configure Firewall
nfs-haproxy
sudo firewall-cmd --permanent --add-service={http,https,nfs,rpc-bind,mountd}
sudo firewall-cmd --permanent --add-port={6443/tcp,22623/tcp}
sudo firewall-cmd --reload
Configure NFS Export
nfs-haproxy
sudo mkdir -p /srv/nfs/ocp-registry
sudo chown -R nobody:nobody /srv/nfs/ocp-registry
sudo chmod 777 /srv/nfs/ocp-registry

# Format second disk
sudo mkfs.xfs /dev/sdb
sudo mount /dev/sdb /srv/nfs/ocp-registry
echo "/dev/sdb /srv/nfs/ocp-registry xfs defaults 0 0" | sudo tee -a /etc/fstab

# Export
echo "/srv/nfs/ocp-registry *(rw,sync,no_subtree_check,no_root_squash)" | sudo tee -a /etc/exports
sudo exportfs -r
Verify

showmount -e localhost — shows /srv/nfs/ocp-registry *

Configure nginx for Ignition
nfs-haproxy
sudo mkdir -p /usr/share/nginx/html/ignition
sudo systemctl restart nginx
curl -I http://localhost/ignition/
Verify

curl -I http://172.20.0.210/ignition/ returns HTTP 200/403 .

SELinux Configuration
nfs-haproxy
sudo setsebool -P haproxy_connect_any on
sudo setsebool -P nfs_export_all_rw on
sudo semanage fcontext -a -t httpd_sys_content_t "/usr/share/nginx/html/ignition(/.*)?"
sudo restorecon -Rv /usr/share/nginx/html/ignition

OpenShift VMs

All commands run on WSL . Do not power on yet — ignition must be injected first ( Section 7 ) .

Clone Bootstrap
WSL
govc vm.clone \
  -vm rhcos-4.14-template -name bootstrap \
  -net "VM Network" -net.adapter vmxnet3 \
  -folder /Datacenter/vm/ \
  -pool /Datacenter/host/Openshift/Resources \
  -ds hdd -on false

govc vm.change -vm bootstrap -m 16384 -c 4
govc vm.disk.change -vm bootstrap -disk.label "Hard disk 1" -size "120G"
Clone Master Nodes
WSL
for i in 0 1 2; do
  govc vm.clone \
    -vm rhcos-4.14-template -name master-$i \
    -net "VM Network" -net.adapter vmxnet3 \
    -folder /Datacenter/vm/ \
    -pool /Datacenter/host/Openshift/Resources \
    -ds hdd -on false
  govc vm.change -vm master-$i -m 16384 -c 4
  govc vm.disk.change -vm master-$i -disk.label "Hard disk 1" -size "120G"
done
Clone Worker Nodes
WSL
for i in 0 1; do
  govc vm.clone \
    -vm rhcos-4.14-template -name worker-$i \
    -net "VM Network" -net.adapter vmxnet3 \
    -folder /Datacenter/vm/ \
    -pool /Datacenter/host/Openshift/Resources \
    -ds hdd -on false
  govc vm.change -vm worker-$i -m 8192 -c 2
  govc vm.disk.change -vm worker-$i -disk.label "Hard disk 1" -size "120G"
done
Verify

govc ls /Datacenter/vm/{bootstrap,master-{0,1,2},worker-{0,1}} — 6 VMs listed .

SCP Ignition to nfs-haproxy

Inject Ignition into VMs
WSL
BOOTSTRAP_IGN_B64=$(base64 -w0 $OCP4_DIR/config/bootstrap.ign)
MASTER_IGN_B64=$(base64 -w0 $OCP4_DIR/config/master.ign)
WORKER_IGN_B64=$(base64 -w0 $OCP4_DIR/config/worker.ign)

govc vm.change -vm bootstrap \
  -e "guestinfo.ignition.config.data=$BOOTSTRAP_IGN_B64" \
  -e "guestinfo.ignition.config.data.encoding=base64"

for i in 0 1 2; do
  govc vm.change -vm master-$i \
    -e "guestinfo.ignition.config.data=$MASTER_IGN_B64" \
    -e "guestinfo.ignition.config.data.encoding=base64"
done

for i in 0 1; do
  govc vm.change -vm worker-$i \
    -e "guestinfo.ignition.config.data=$WORKER_IGN_B64" \
    -e "guestinfo.ignition.config.data.encoding=base64"
done
Verify

govc vm.info -e bootstrap | grep guestinfo.ignition shows both keys .

Cluster Install

Power On Bootstrap
WSL
govc vm.power -on bootstrap
openshift-install wait-for bootstrap-complete \
  --dir=$OCP4_DIR/config --log-level debug
ℹ
This typically takes 10-15 minutes . Ends with It is now safe to remove the bootstrap resources .
Power On Masters
WSL
for i in 0 1 2; do
  govc vm.power -on master-$i
  echo "master-$i powered on"
done
Approve CSRs
WSL
export KUBECONFIG=$OCP4_DIR/config/auth/kubeconfig

# Auto-approve pending CSRs
oc get csr -o go-template='{{range .items}}{{if not .status}}{{.metadata.name}}{{"\n"}}{{end}}{{end}}' | xargs oc adm certificate approve

# Monitor in loop
watch -n 10 'oc get csr -o go-template="{{range .items}}{{if not .status}}{{.metadata.name}} {{end}}{{end}}" | xargs -r oc adm certificate approve'
Verify

oc get nodes shows masters as Ready

Power On Workers
WSL
govc vm.power -on worker-0
govc vm.power -on worker-1

# Approve worker CSRs
oc get csr -o go-template='{{range .items}}{{if not .status}}{{.metadata.name}}{{"\n"}}{{end}}{{end}}' | xargs oc adm certificate approve
Wait for Install Complete
WSL
# Can take 30-60 minutes
openshift-install wait-for install-complete \
  --dir=$OCP4_DIR/config --log-level debug
Verify

Output ends with web console URL and admin credentials .

oc get nodes shows all 5 nodes ( 3 masters + 2 workers ) as Ready

Remove Bootstrap
WSL
govc vm.power -off bootstrap
govc vm.destroy bootstrap

ssh -i ~/.ssh/openshift khaled@172.20.0.210 \
  sudo sed -i '/server bootstrap/d' /etc/haproxy/haproxy.cfg && \
  sudo systemctl restart haproxy

Cleanup

Destroy all VMs in reverse order . All commands on WSL .

WSL — Destroy all VMs
# Workers
govc vm.power -off worker-0 worker-1
govc vm.destroy worker-0 worker-1

# Masters
govc vm.power -off master-0 master-1 master-2
govc vm.destroy master-0 master-1 master-2

# Bootstrap (if still exists)
govc vm.power -off bootstrap 2>/dev/null; govc vm.destroy bootstrap 2>/dev/null

# nfs-haproxy (skip if you want to keep it)
govc vm.power -off nfs-haproxy
govc vm.destroy nfs-haproxy

# Template
govc vm.destroy rhcos-4.14-template

# Verify empty
govc ls /Datacenter/vm/
