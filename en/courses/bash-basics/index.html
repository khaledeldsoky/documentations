Bash Scripting Beginner
1

Check if a File Exists

file existence check 
#!/bin/bash
file="test.txt"
if [ -f "$file" ]; then
    echo "$file exists"
else
    echo "$file does not exist"
fi
2

Compare Numbers

numeric comparison 
#!/bin/bash
num=15
if [ $num -gt 10 ]; then
    echo "$num is greater than 10"
elif [ $num -eq 10 ]; then
    echo "$num equals 10"
else
    echo "$num is less than 10"
fi
3

Compare Strings

string comparison 
#!/bin/bash
user="Alice"
if [ "$user" = "Alice" ]; then
    echo "Welcome, $user!"
else
    echo "Unknown user"
fi
4

Check if Directory is Empty

empty directory check 
#!/bin/bash
dir="myfolder"
if [ -z "$(ls -A $dir)" ]; then
    echo "$dir is empty"
else
    echo "$dir is not empty"
fi
5

For Loop — Iterate Over a List

iterate over words 
#!/bin/bash
for fruit in apple banana orange; do
    echo "Fruit: $fruit"
done
6

For Loop — Loop Over Files

loop over files in directory 
#!/bin/bash
for file in *.txt; do
    echo "Found file: ${file}"
done
list all files in current dir 
#!/bin/bash
for file in *; do
    echo "File: $file"
done
7

For Loop — Numeric Range

iterate over a range 
#!/bin/bash
for i in {1..5}; do
    echo "Number: $i"
done
8

While Loop — Count

count from 1 to 5 
#!/bin/bash
count=1
while [ $count -le 5 ]; do
    echo "Count: $count"
    ((count++))
done
print numbers 1 to 10 
num=1
while [ $num -le 10 ]; do
    echo "$num"
    ((num++))
done
9

While Loop — Read File

read file line by line 
#!/bin/bash
file="input.txt"
while IFS= read -r line; do
    echo "Line: $line"
done < "$file"
10

While Loop — Infinite with User Input

infinite loop until 'quit' 
#!/bin/bash
while true; do
    echo "Enter 'quit' to exit: "
    read input
    if [ "$input" = "quit" ]; then
        break
    fi
    echo "You entered: $input"
done

Functions

define and call a function
