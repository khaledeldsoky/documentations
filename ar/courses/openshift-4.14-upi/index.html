بيئة vSphere

تنزيل RHCOS OVA
WSL
export RHCOS_VERSION=4.14.0
cd $OCP4_DIR/rhcos
wget https://mirror.openshift.com/pub/openshift-v4/dependencies/rhcos/4.14/${RHCOS_VERSION}/rhcos-${RHCOS_VERSION}-x86_64-vmware.x86_64.ova
sha256sum rhcos-${RHCOS_VERSION}-x86_64-vmware.x86_64.ova
استيراد RHCOS OVA كـ Template
WSL
govc import.ova \
  -name rhcos-4.14-template \
  -ds hdd \
  -pool /Datacenter/host/Openshift/Resources \
  -folder /Datacenter/vm/ \
  $OCP4_DIR/rhcos/rhcos-4.14.0-x86_64-vmware.x86_64.ova

govc vm.markastemplate rhcos-4.14-template

جهاز HAProxy + NFS

إنشاء الجهاز من ISO
WSL
govc vm.create \
  -name nfs-haproxy \
  -m 8192 -c 4 \
  -disk "40G" \
  -disk.controller pvscsi \
  -net "VM Network" \
  -pool /Datacenter/host/Openshift/Resources \
  -folder /Datacenter/vm/ \
  -on false

govc vm.dvd.insert \
  -vm nfs-haproxy \
  -ds hdd \
  /hdd/isos/CentOS-Stream-9-latest-x86_64-dvd1.iso

govc vm.disk.create \
  -vm nfs-haproxy \
  -name nfs-haproxy/nfs-data \
  -size "100G" -ds hdd -thin

govc vm.power -on nfs-haproxy
govc vm.console nfs-haproxy

تكوين HAProxy
nfs-haproxy — /etc/haproxy/haproxy.cfg
global
    log /dev/log local0
    chroot /var/lib/haproxy
    stats socket /run/haproxy/admin.sock mode 660
    user haproxy group haproxy
    daemon

defaults
    log global
    mode tcp
    option tcplog
    timeout connect 10s
    timeout client 30s
    timeout server 30s

frontend openshift-api
    bind *:6443
    default_backend openshift-api

backend openshift-api
    balance source
    server bootstrap 172.20.0.230:6443 check
    server master-0 172.20.0.231:6443 check
    server master-1 172.20.0.232:6443 check
    server master-2 172.20.0.233:6443 check

frontend mcs
    bind *:22623
    default_backend mcs

backend mcs
    balance source
    server bootstrap 172.20.0.230:22623 check
    server master-0 172.20.0.231:22623 check
    server master-1 172.20.0.232:22623 check
    server master-2 172.20.0.233:22623 check

frontend openshift-ingress
    bind *:443
    bind *:80
    default_backend openshift-ingress

    server worker-0 172.20.0.234:443 check
    server worker-1 172.20.0.235:443 check
nfs-haproxy — تطبيق الإعدادات
sudo haproxy -c -f /etc/haproxy/haproxy.cfg
sudo systemctl restart haproxy
sudo ss -tlnp | grep -E ':(6443|22623|443|80)'
تكوين الجدار الناري
nfs-haproxy

sudo firewall-cmd --reload
تكوين مشاركة NFS
nfs-haproxy
sudo mkdir -p /srv/nfs/ocp-registry
sudo chown -R nobody:nobody /srv/nfs/ocp-registry
sudo chmod 777 /srv/nfs/ocp-registry

sudo mkfs.xfs /dev/sdb
sudo mount /dev/sdb /srv/nfs/ocp-registry
echo "/dev/sdb /srv/nfs/ocp-registry xfs defaults 0 0" | sudo tee -a /etc/fstab

echo "/srv/nfs/ocp-registry *(rw,sync,no_subtree_check,no_root_squash)" | sudo tee -a /etc/exports
sudo exportfs -r

sudo restorecon -Rv /usr/share/nginx/html/ignition

أجهزة OpenShift

استنساخ Bootstrap
WSL
govc vm.clone \
  -vm rhcos-4.14-template -name bootstrap \
  -net "VM Network" -net.adapter vmxnet3 \
  -folder /Datacenter/vm/ \
  -pool /Datacenter/host/Openshift/Resources \
  -ds hdd -on false

govc vm.change -vm bootstrap -m 16384 -c 4
govc vm.disk.change -vm bootstrap -disk.label "Hard disk 1" -size "120G"
استنساخ Master
WSL
for i in 0 1 2; do
  govc vm.clone \
    -vm rhcos-4.14-template -name master-$i \
    -net "VM Network" -net.adapter vmxnet3 \
    -folder /Datacenter/vm/ \
    -pool /Datacenter/host/Openshift/Resources \
    -ds hdd -on false
  govc vm.change -vm master-$i -m 16384 -c 4
  govc vm.disk.change -vm master-$i -disk.label "Hard disk 1" -size "120G"
done
استنساخ Worker
WSL
for i in 0 1; do
  govc vm.clone \
    -vm rhcos-4.14-template -name worker-$i \
    -net "VM Network" -net.adapter vmxnet3 \
    -folder /Datacenter/vm/ \
    -pool /Datacenter/host/Openshift/Resources \
    -ds hdd -on false
  govc vm.change -vm worker-$i -m 8192 -c 2
  govc vm.disk.change -vm worker-$i -disk.label "Hard disk 1" -size "120G"
done

حقن Ignition في الأجهزة الظاهرية
WSL
BOOTSTRAP_IGN_B64=$(base64 -w0 $OCP4_DIR/config/bootstrap.ign)
MASTER_IGN_B64=$(base64 -w0 $OCP4_DIR/config/master.ign)
WORKER_IGN_B64=$(base64 -w0 $OCP4_DIR/config/worker.ign)

govc vm.change -vm bootstrap \
  -e "guestinfo.ignition.config.data=$BOOTSTRAP_IGN_B64" \
  -e "guestinfo.ignition.config.data.encoding=base64"

for i in 0 1 2; do
  govc vm.change -vm master-$i \
    -e "guestinfo.ignition.config.data=$MASTER_IGN_B64" \
    -e "guestinfo.ignition.config.data.encoding=base64"
done

for i in 0 1; do
  govc vm.change -vm worker-$i \
    -e "guestinfo.ignition.config.data=$WORKER_IGN_B64" \
    -e "guestinfo.ignition.config.data.encoding=base64"
done
